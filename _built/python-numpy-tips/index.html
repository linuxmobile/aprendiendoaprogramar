    Count 0, NaNs 

    # count nans
    arr = [1,2,3,4, np.nan,np.nan, 0,0,0]
    np.isnan(arr).sum() # 2
    # count non-nans
    arr = [1,2,3,4, np.nan,np.nan, 0,0,0]
    np.count_nonzero(~np.isnan(arr)) # 7
    # count non-zeros
    arr = [1,2,3,4, np.nan,np.nan, 0,0,0]
    np.count_nonzero(arr) # 6

    Deleting 

    # DELETE BY POSITIONS
    arr = np.arange(6)
    np.delete(arr, [3,4])
    array([0, 1, 2, 3, 4, 5])
array([0, 1, 2, 5])

    # DELETE BY VALUES
    a = np.array([1, 2, 3, 4, 5, 6])
    np.delete(a, np.where(a == 3))

    b = np.array([1, 2, 3, 'a'])
    np.delete(a, np.where(b == 'a'))

    # OR
    a[a!=3]
    b[b!='a']
    array([1, 2, 4, 5, 6])
